name: GitHub Pages

  push:
    branches:
      - master

jobs:
  build-deploy:
    runs-on: ubuntu-22.04
    concurrency:
      group: ${{ github.workflow }}-${{ github.ref }}
    steps:
      - uses: actions/checkout@v3

      - name: Setup Hugo

          hugo-version: 'latest'
          extended: true

      - name: Build
        run: hugo --minify --config vps-deploy.toml
           
      - name: rsync deployments

          switches: -avzr --delete
